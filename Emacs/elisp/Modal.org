* View mode for modal editing

#+begin_src emacs-lisp
(require 'view)

(defun view-mode-all-files ()
  (interactive)
  (let ((state (not view-mode)))
    (delq 'nil
          (mapcar (lambda (file)
                    (when
                        (and (buffer-file-name file)
                             (file-exists-p (buffer-file-name file)))
                      (with-current-buffer (buffer-name file)
                        (if state
                            (view-mode)
                          (View-exit)))
                      (buffer-name file)))
                  (buffer-list)))))

#+end_src
